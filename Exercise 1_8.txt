(define (square x)
  (* x x))
(define (average x y)
  (/ (+ x y) 2))
(define (improve guess x)
  (/ (+ (/ x (square guess))
        (* 2 guess))
     3))
(define (good-enough? guess x old-guess)
  (< (abs (/ (- old-guess guess) guess)) 0.001))
(define (cbrt-iter guess x old-guess)
  (if (good-enough? guess x old-guess)
  guess
  (cbrt-iter (improve guess x)
    x guess)))
(define (cbrt x)
  (cbrt-iter 1.0 x 0.3))
